\section{Funcionales}
\vspace{-0.5em}
\begin{longtable}{|p{3cm}|p{10cm}|}
	\hline
	\textbf{RF001} & \textbf{Guía visual en tiempo real} \\
	\hline
     \textbf{Descripción:} & Mostrar flechas en el entorno real para dirigir al alumno al salón destino. \\
 	\hline
	\textbf{RF002} & \textbf{Mapas interactiivos} \\
	\hline
     \textbf{Descripción:} & Proporcionar un mapa 3D del edificio que el usuario pueda consultar en cualquier momento, con su posición actual marcada. \\
     \hline
	\textbf{RF003} & \textbf{Detección de ubicación en interiores} \\
	\hline
     \textbf{Descripción:} & Usar realidad aumentada para determinar la ubicación del usuario dentro del edificio. \\
     \hline
    \textbf{RF004} & \textbf{Busqueda de salones} \\
    \hline
    \textbf{Descripción:} & Permitir al usuario buscar un salón específico. \\
    \hline
    \textbf{RF005} & \textbf{Sincronización de horario} \\
    \hline
    \textbf{Descripción:} & Conectar a una API o base de datos para mostrar horarios de clases \\
    \hline
    \textbf{RF006} & \textbf{Historial de rutas} \\
    \hline
    \textbf{Descripción:} & Guardar las rutas previamente consultadas por el usuario para acceso rápido. \\
    \hline
    \textbf{RF007} & \textbf{Modo offline} \\
    \hline
    \textbf{Descripción:} & Permitir el uso de la aplicación sin conexión a internet, descargando previamente los datos necesarios. \\
    \hline
    \textbf{RF008} & \textbf{Retroalimentación visual} \\
    \hline
    \textbf{Descripción:} & Proporcionar indicaciones visuales claras y comprensibles para el usuario durante la navegación. \\
    \hline
    \textbf{RF009} & \textbf{Registro de estudiantes} \\
    \hline
    \textbf{Descripción:} & Permitir a los estudiantes registrarse en la aplicación para personalizar su experiencia. \\
    \hline
\end{longtable}
